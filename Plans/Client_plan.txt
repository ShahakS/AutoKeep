Login:

    {
        ProtocolMessage : "LOGIN" ,
        user : {[
            emailAddress : "user name" ,
            password : "password"
            }]
    }

    Server send : ProtocolMessage , user type , result
    {
        ProtocolMessage = "login"
        user_type : "user" | "admin",
        result = "ok" | "wrong"
    }
