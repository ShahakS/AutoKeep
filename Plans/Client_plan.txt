Login:
    Client send : ProtocolMessage , emailAddress , password 
    {
        ProtocolMessage : "LOGIN" ,
        LOGIN : {[
            emailAddress : "" ,
            password : ""
            }]
    }

    Server send : ProtocolMessage , user_type , result
    {
        ProtocolMessage = "LOGIN_RESULTS"
        user_type : "user" | "admin",
        result = "ok" | "wrong"
    }
